import actions from '../../actions';
 
const {
  ADD_DATA_OBJECT
} = actions;
 
// const {
//   createAddDataObjectAction
//   // ,
//   // createBlobObjectAction,
//   // createCommitObjectAction,
//   // createRefObjectAction
// } = actions;
 
export const defaultState = {
  objects: {}
};
 
export const objects = (state = defaultState, action) => {
 
  let newState = {
    ...state
  };
 
  const { type, payload } = action;
  let hash;
 
  switch (type) {
 
  case ADD_DATA_OBJECT:
 
    hash = payload.hash;
 
    if (!hash) {
      break;
    }
 
    newState = {
      ...newState,
      objects: {
        ...newState.objects,
        [hash.substr(0, 2)]: {
          [hash.substr(2)]: {
            data: payload.data
          }
        }
      }
    };
 
    break;
 
  default:
    break;
  }
 
  return newState;
 
};
 
export default objects;
